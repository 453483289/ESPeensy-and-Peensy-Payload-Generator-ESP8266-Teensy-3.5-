void StartFromSD (String Filename){
    File myFile;
    String line;
    // Serial.println ("Init SD");
    if (SD.begin(BUILTIN_SDCARD)) {
      if (SD.exists("1.txt")) {
        myFile = SD.open("1.txt");
          if (myFile) {
            while (myFile.available()) {
              line = myFile.readStringUntil('\n');
              RunDuckyLine (line);
            }
            myFile.close();
          }
      }
  }
}
